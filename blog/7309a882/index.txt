export default async function fetchBooks(q?:string) : Promise<BookData[]> {
  let url = 'http://localhost:12345/book'
  if( q ) {
    url += `/search?q=${q}`
  }
  try {
    const response = await fetch(url)
    if (response.ok) {
      const books = await response.json()
      return books
    } else {
      throw new Error()
    }
  } catch (error) {
    console.error(error)
    return []
  }
}

  const q = context.query.q;

  }
}
